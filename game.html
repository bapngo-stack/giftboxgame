        // Phát nhạc nền khi trang tải

        window.onload = function () {

        function showLetters(giftNumber) {

            const container = document.getElementById('game-container');

            const lettersContainer = document.getElementById('letters-container');

            const backButton = document.getElementById('back-button');


            // Ẩn tất cả các hộp quà

            document.querySelectorAll('.gift-box').forEach(box => {

                box.style.display = 'none';

            });


            // Hiển thị container chứa lá thư và nút quay lại

            lettersContainer.style.display = 'flex';

            lettersContainer.innerHTML = '';

            backButton.style.display = 'block';

            

            


            const messages = [

                'Chúc bạn một ngày thật vui vẻ và tràn đầy năng lượng!',

                'Hãy luôn tin vào khả năng của bản thân!',

                'Mỗi ngày là một cơ hội mới để trưởng thành.',

                'Cảm ơn bạn đã luôn kiên nhẫn và mạnh mẽ!',

                'Chúc bạn thành công trong mọi dự định!',

                'Hãy giữ vững ước mơ và đừng bao giờ từ bỏ!',

                'Sự chân thành là món quà quý giá nhất.',

                'Hãy yêu thương và sẻ chia mọi khoảnh khắc.',

                'Bạn là người đặc biệt và đáng trân trọng.',

                'Chúc bạn luôn tìm thấy hạnh phúc trong cuộc sống.'

            ];


            for (let i = 0; i < messages.length; i++) {

                const letter = document.createElement('div');

                letter.classList.add('letter');

                letter.textContent = `Lá thư ${giftNumber}-${i + 1}`;


                letter.onclick = () => {

                    letter.style.display = 'none';

                    showMessage(messages[i]);

                };


                lettersContainer.appendChild(letter);

            }

        }


        function showMessage(messageContent) {

            const lettersContainer = document.getElementById('letters-container');

            const message = document.getElementById('message');


            lettersContainer.style.display = 'none';

            message.textContent = messageContent;

            message.style.display = 'block';

        }


        function goBack() {

            const message = document.getElementById('message');

            const lettersContainer = document.getElementById('letters-container');

            const backButton = document.getElementById('back-button');


            message.style.display = 'none';

            lettersContainer.style.display = 'none';

            backButton.style.display = 'none';


            // Hiển thị lại các hộp quà

            document.querySelectorAll('.gift-box').forEach(box => {

                box.style.display = 'block';

            });

        }

function closeInstructions() {

            const instructions = document.getElementById('instructions');

            instructions.style.display = 'none';

        }

        function shakeBox(box) {

            box.style.transform = 'scale(1.1)';

            setTimeout(() => {

                box.style.transform = 'scale(1)';

            }, 300);

        }
